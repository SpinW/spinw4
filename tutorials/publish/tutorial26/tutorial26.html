Q = ch.mag_str.k(1);

J = @(k)2*J1*cos(2*pi*k)+2*J2*cos(4*pi*k);
A = @(k)J(k)+J(k+Q)/2+J(k-Q)/2-2*J(Q);
B = @(k)J(k)-J(k+Q)/2-J(k-Q)/2;

k = linspace(0,1,500);

w1 =  @(k)sqrt(A(k).^2-B(k).^2);
w2 =  @(k)-sqrt(A(k).^2-B(k).^2);
%w1 = A(k)+B(k);

Sxixi = @(k)2*(A(k)+B(k))./w1(k);
